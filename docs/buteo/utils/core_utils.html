 538def get_augmented_path(
 539    path: str,
 540    *,
 541    prefix: str = "",
 542    suffix: str = "",
 543    add_uuid: bool = True,
 544    folder: Optional[str] = None,
 545) -> str:
 546    """
 547    Gets a basename from a string in the format:
 548    `dir/prefix_basename_time_uuid_suffix.ext`
 549
 550    Args:
 551        path (str): The path to the original file.
 552
 553    Keyword Args:
 554        prefix (str = ""): The prefix to add to the memory path.
 555        suffix (str = ""): The suffix to add to the memory path.
 556        add_uuid (bool = True): If True, add a uuid to the memory path.
 557        folder (str = None): The folder to save the file in.
 558
 559    Returns:
 560        str: A string of the current UNIX time in seconds.
 561    """
 562    assert isinstance(path, str), "path must be a string."
 563    assert isinstance(folder, (type(None), str)), "folder must be None or a string"
 564    assert len(os.path.splitext(os.path.basename(path))[1]) > 1, f"Path must have an extension. {path}"
 565
 566    if os.path.basename(os.path.abspath(path)) == path:
 567        path = PurePosixPath("/vsimem", path).as_posix()
 568
 569    if folder is not None:
 570
 571        if folder.startswith("/vsimem"):
 572            path = PurePosixPath("/vsimem", os.path.basename(path)).as_posix()
 573        else:
 574            assert folder_exists(folder), f"folder must exist. {folder}"
 575            path = os.path.join(folder, os.path.basename(path))
 576
 577        if not path.startswith(f"{os.sep}vsimem") and not folder_exists(path_to_folder(folder)):
 578            raise ValueError(f"Folder: {folder} does not exist.")
 579
 580    assert is_valid_file_path(path), f"path must be a valid file path. {path}"
 581
 582    base = os.path.basename(path)
 583    split = list(os.path.splitext(base))
 584
 585    uuid = ""
 586    if add_uuid:
 587        uuid = f"_{get_unix_seconds_as_str()}_{str(uuid4())}"
 588
 589    if is_valid_mem_path(path):
 590        if split[1][1:] in get_valid_raster_driver_extensions():
 591            split[1] = ".tif"
 592        elif split[1][1:] in get_valid_vector_driver_extensions():
 593            split[1] = ".gpkg"
 594        else:
 595            raise ValueError("Unable to parse file extension as valid datasource.")
 596
 597    basename = f"{prefix}{split[0]}{uuid}{suffix}{split[1]}"
 598
 599    if is_valid_mem_path(path):
 600        out_path = PurePosixPath("/vsimem", basename).as_posix()
 601    else:
 602        out_path = os.path.join(os.path.dirname(os.path.abspath(path)), basename)
 603
 604    return out_path

 639def divide_into_steps(
 640    total: int,
 641    step: int,
 642):
 643    """
 644    Divide a number into steps.
 645
 646    Args:
 647        total int: The total number.

 649
 650    Returns:
 651        list: The list of steps.
 652    """
 653    assert isinstance(total, int), "total must be an integer."
 654    assert isinstance(step, int), "step must be an integer."
 655
 656    steps = []
 657    remainder = total % step
 658    divided = int(total / step)
 659    for _ in range(step):
 660        if remainder > 0:
 661            steps.append(divided + 1)
 662            remainder -= 1
 663        else:
 664            steps.append(divided)
 665
 666    return steps

 759def type_check(
 760    variable: Any,
 761    types: Tuple[type, ...],
 762    name: str = "",
 763    *,
 764    throw_error: bool = True,
 765) -> bool:
 766    """
 767    Utility function to type check the inputs of a function. Check two levels down.
 768
 769    Args:
 770        variable (any): The variable to check.
 771        types (tuple): The types to check against. (float, int, ...)
 772
 773    Keyword Args:
 774        name (str = ""): The name printed in the error string if an error is thrown.
 775        throw_error (bool = True): If True, raise an error if the type is not correct.
 776
 777    Returns:
 778        bool: A boolean indicating if the type is valid. If throw_error an error is raised if the input is not a valid type.
 779    """
 780    assert isinstance(name, str), "name must be a string."
 781    assert recursive_check_type_list_none_or_tuple(types), f"types must be a type, list, None, or tuple. not: {types}"
 782
 783    if not isinstance(types, (list, tuple)):
 784        types = [types]
 785
 786    valid_types = []
 787    for valid_type in types:
 788        if valid_type is None:
 789            valid_types.append(type(None))
 790        elif isinstance(valid_type, type):
 791            valid_types.append(valid_type)
 792        elif isinstance(valid_type, (list, tuple)):
 793            valid_types.append(valid_type)
 794        else:
 795            raise ValueError(f"Invalid type: {valid_type}")
 796
 797    if not isinstance(variable, (list, tuple)):
 798        sublist_valid_types = []
